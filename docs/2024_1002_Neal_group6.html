pvalue=numeric(); k=0
for (i in 1:length(continuous_var))
{
  group_6A=finalized_data %>% filter(Month_Group=="6A") %>% select(continuous_var[i]) %>% pull()
  group_6B=finalized_data %>% filter(Month_Group=="6B") %>% select(continuous_var[i]) %>% pull()
  group_6C=finalized_data %>% filter(Month_Group=="6C") %>% select(continuous_var[i]) %>% pull()
  
  pvalue[(k-1)*3+1]=t.test(group_6A, group_6B)$p.value
  pvalue[(k-1)*3+2]=t.test(group_6A, group_6C)$p.value
  pvalue[(k-1)*3+3]=t.test(group_6B, group_6C)$p.value
  
}
pvalue_df <- map_dfr(continuous_var, function(var) {
  group_data <- finalized_data %>%
    select(Month_Group, value = all_of(var)) %>%
    filter(!is.na(value))
  
  group_6A <- filter(group_data, Month_Group == "6A") %>% pull(value)
  group_6B <- filter(group_data, Month_Group == "6B") %>% pull(value)
  group_6C <- filter(group_data, Month_Group == "6C") %>% pull(value)

  tibble(
    Variable = var,
    Comparison = c("6A vs 6B", "6A vs 6C", "6B vs 6C"),
    p_value = c(
      adjust_pvalue(t.test(group_6A, group_6B)$p.value),
      adjust_pvalue(t.test(group_6A, group_6C)$p.value),
      adjust_pvalue(t.test(group_6B, group_6C)$p.value)
    )
  )
})
