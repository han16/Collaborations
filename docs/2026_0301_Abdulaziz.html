analyze_dental_color <- function(data, target_type, target_material) {
  
  # 1. Filter and Prepare Data
  analysis_df <- data %>%
    filter(`...4` == target_type & material == target_material) %>%
    mutate(specimen_id = str_extract(...1, "\\d$"))
  
  # 2. Repeated Measures ANOVA
  anova_table <- analysis_df %>%
    group_by(solution) %>%
    anova_test(dv = DeltaE, wid = specimen_id, within = time_interval) %>%
    get_anova_table()
  
  # 3. Pairwise Comparisons
  pairwise_table <- analysis_df %>%
    group_by(solution) %>%
    pairwise_t_test(
      DeltaE ~ time_interval, 
      paired = TRUE, 
      p.adjust.method = "bonferroni",
      conf.level = 0.95,
      detailed = TRUE
    ) %>%
    rename(mean_diff = estimate) %>%
    select(solution, group1, group2, n1, mean_diff, conf.low, conf.high, p.adj, p.adj.signif) %>%
    # Rounding the numerical columns
    mutate(
      mean_diff = round(mean_diff, 4),
      conf.low = round(conf.low, 4),
      conf.high = round(conf.high, 4)
    )
  
  # Return both tables as a list
  return(list(anova = anova_table, pairwise = pairwise_table))
}
compare_all_scenarios <- function(data, target_type, mat1, mat2) {
  
  # 1. Filter for the two materials and the specific measurement type (SCI/SCE)
  comparison_df <- data %>%
    filter(`...4` == target_type) %>%
    filter(material %in% c(mat1, mat2))
  
  # 2. Group by solution and interval, then run independent t-tests
  results_table <- comparison_df %>%
    group_by(solution, time_interval) %>%
    t_test(
      DeltaE ~ material, 
      paired = FALSE,         # Materials are separate specimens
      detailed = TRUE,
      conf.level = 0.95
    ) %>%
    # 3. Add significance thresholds
    add_significance(
      p.col = "p",
      cutpoints = c(0, 0.001, 0.01, 0.05, 1),
      symbols = c("***", "**", "*", "ns")
    ) %>%
    # 4. Clean up, rename, and format
    rename(mean_diff = estimate) %>%
    mutate(
      across(c(mean_diff, conf.low, conf.high), ~ round(.x, 4)),
      p = format.pval(p, digits = 4, eps = 0.001)
    ) %>%
    # 5. Select key columns for comparison
    select(solution, time_interval, group1, group2, n1, n2, mean_diff, conf.low, conf.high, p, p.signif)
  
  return(results_table)
}
