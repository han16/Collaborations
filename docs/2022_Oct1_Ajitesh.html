library(tidyverse)
library(DT)
library(htmltools)

# List to store the tables
all_tables <- list()

numeric_vars <- colnames(trueness)[3:11]
methods2 <- unique(trueness$methods)

for (i in seq_along(numeric_vars)) {
  var_name <- numeric_vars[i]
  
  # 1. Generate combinations and perform tests
  stats <- expand.grid(group1 = methods2, group2 = methods2, stringsAsFactors = FALSE) %>%
    rowwise() %>%
    mutate(
      sample1 = list(trueness %>% filter(methods == group1) %>% pull(all_of(var_name))),
      sample2 = list(trueness %>% filter(methods == group2) %>% pull(all_of(var_name))),
      test = list(t.test(sample1, sample2)),
      mean_diff = round(test$estimate[1] - test$estimate[2], 2),
      CI_lower = round(test$conf.int[1], 2),
      CI_upper = round(test$conf.int[2], 2),
      p_raw = test$p.value
    ) %>%
    ungroup() %>%
    mutate(p_adj = p.adjust(p_raw, method = "bonferroni")) %>%
    mutate(p_adj = ifelse(p_adj < 0.001, "< 0.001", round(p_adj, 4)))
  
  # 2. Formatting
  final_table <- stats %>%
    dplyr::select(group1, group2, mean_diff, CI_lower, CI_upper, p_raw, p_adj) 
  
  # 3. Create the DT object with a unique ID
  # We wrap it in a tagList item
  all_tables[[i]] <- htmltools::tags$div(
    htmltools::tags$h3(paste("Variable:", var_name)),
    datatable(final_table, 
              extensions = 'Buttons',
              elementId = paste0("table_", i), # Critical for HTML rendering in loops
              options = list(
                dom = 'Blfrtip',
                buttons = c('copy', 'csv', 'excel', 'pdf', 'print'),
                lengthMenu = list(c(10, 25, 50, -1), c(10, 25, 50, "All"))
              )
    ),
    htmltools::tags$br()
  )
}

# This is the "magic" line that prints everything to the HTML
htmltools::tagList(all_tables)

# Container for all HTML tables

# Loop through each element in the list (1 to 9)
for (k in 1:length(precision_data_all)) {
  
  df_current <- precision_data_all[[k]]
  methods3 <- unique(df_current$methods)
  
  # 1. Generate all pairwise combinations
  stats <- expand.grid(group1 = methods3, group2 = methods3, stringsAsFactors = FALSE) %>%
    rowwise() %>%
    mutate(
      sample1 = list(df_current %>% filter(methods == group1) %>% pull(vs_distance)),
      sample2 = list(df_current %>% filter(methods == group2) %>% pull(vs_distance)),
      test = list(t.test(sample1, sample2)), # Non-paired t-test
      mean_difference = round(test$estimate[1] - test$estimate[2], 2),
      CI_lower = round(test$conf.int[1], 2),
      CI_upper = round(test$conf.int[2], 2),
      p_raw = test$p.value
    ) %>%
    ungroup()
  
  # 2. Apply p-value adjustment (Bonferroni)
  stats <- stats %>%
    mutate(p_adj = p.adjust(p_raw, method = "bonferroni")) %>%
    mutate(p_adj = ifelse(p_adj < 0.001, "< 0.001", round(p_adj, 4)))
  
  # 3. Clean up repetitive group names for visual clarity
  final_table <- stats %>%
    dplyr::select(group1, group2, mean_difference, CI_lower, CI_upper, p_raw, p_adj)
