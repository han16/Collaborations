library(tidyverse)
library(blandr)

compare_methods_ba <- function(df, method_a, method_b) {
  
  # 1. Identify numeric variables to analyze
  vars_to_analyze <- df %>% 
    select(where(is.numeric), -Implant.position) %>% 
    colnames()
  
  # 2. Create a standardized 'Match_ID' based on the trailing number
  # This turns "FGAS_LS_1" and "FGAS_IOS_1" both into "1"
  df_clean <- df %>%
    filter(methods %in% c(method_a, method_b)) %>%
    mutate(Match_ID = as.numeric(str_extract(Sample, "\\d+$")))
  
  # 3. Pivot the data using our new Match_ID
  df_wide <- df_clean %>%
    select(Match_ID, methods, all_of(vars_to_analyze)) %>%
    pivot_wider(
      names_from = methods,
      values_from = all_of(vars_to_analyze)
    ) %>%
    drop_na() # Ensures we only plot samples present in both groups
  
  # 4. Generate the plots
  for (var in vars_to_analyze) {
    col_a <- paste0(var, "_", method_a)
    col_b <- paste0(var, "_", method_b)
    
    # Using try() to ensure the loop continues if one plot fails
    try({
      p <- blandr.draw(df_wide[[col_a]], df_wide[[col_b]]) +
        labs(
          title = paste("Bland-Altman Agreement:", var),
          subtitle = paste(method_a, "vs.", method_b, "| n =", nrow(df_wide)),
          x = "Mean of Measurements",
          y = "Difference (A - B)"
        ) +
        theme_minimal()
      
      print(p)
    })
  }
}

library(tidyverse)
library(blandr)

compare_methods_pairwise <- function(df, value_col = "vs_distance", method_col = "methods") {
  
  # 1. Create a Match_ID to pair the first observation of Method A 
  # with the first observation of Method B, and so on.
  df_indexed <- df %>%
    group_by(across(all_of(method_col))) %>%
    mutate(Match_ID = row_number()) %>%
    ungroup()
  
  # 2. Get a list of unique methods
  unique_methods <- unique(df[[method_col]])
  
  # 3. Generate all unique pairs (e.g., Method1 vs Method2)
  method_pairs <- combn(unique_methods, 2, simplify = FALSE)
  
  # 4. Loop through each pair and generate the analysis
  for (pair in method_pairs) {
    method_a <- pair[1]
    method_b <- pair[2]
    
    # Filter for the two methods and pivot to wide format for subtraction
    df_pair <- df_indexed %>%
      filter(.data[[method_col]] %in% c(method_a, method_b)) %>%
      select(Match_ID, !!sym(method_col), !!sym(value_col)) %>%
      pivot_wider(names_from = !!sym(method_col), values_from = !!sym(value_col)) %>%
      drop_na() # Ensure only paired samples are analyzed
    
    # Generate the Bland-Altman Plot
    # Using try() to prevent the loop from breaking if one pair has issues
    try({
      p <- blandr.draw(df_pair[[method_a]], df_pair[[method_b]]) +
        labs(
          title = "Pairwise Bland-Altman Analysis",
          subtitle = paste(method_a, "vs.", method_b),
          x = "Mean of Measurements",
          y = "Difference (A - B)"
        ) +
        theme_minimal()
      
      print(p)
    })
  }
}
